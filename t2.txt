import discord
import hypixel
from discord.ext import commands

bot = commands.Bot(command_prefix='>')

@bot.event
async def on_ready():
    print("Ready to use!")

@bot.command()
async def ping(ctx):
    await ctx.send('pong')

@bot.command()
async def send(ctx, message):
    channel = bot.get_channel(718088854250323991)
    await channel.send(message)

bot.run('Token')
